^r4/lib/gui.txt

:testdiv
	-7 ( 7 <? )(
		dup "%d 5 / =" print
		dup 5 / "%d " print cr
		1+ ) drop ;

:testmod
	-7 ( 7 <? )(
		dup "%d 4 mod =" print
		dup 4 mod "%d " print cr
		1+ ) drop ;

:mod4t
	dup 31 >> 30 0>> swap over + 3 and swap - ;

:testmod2
	-7 ( 7 <? )(
		dup "%d 4 mod =" print
		dup mod4t "%d " print cr
		1+ ) drop ;

:testsh
	30 ( 37 <? )(
		dup "256 ffffffff %d *>> " print
		256 $fffffff pick2 *>> "=%h" print
		cr
		1+ ) drop ;


:main
	show clrscr
|        testdiv cr
        testmod cr
        testmod2 cr
|        testsh cr
		'exit >esc<
		;

: main ;
