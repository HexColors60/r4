^r4/lib/gui.txt

:testdiv
	-7 ( 7 <? )(
		dup "%d 5 / =" print
		dup 5 / "%d " print cr
		1+ ) drop ;

:test/mod
	-7 ( 7 <? )(
		dup "%d 4 /mod =" print
		dup 4 /mod "%d %d" print cr
		1+ ) drop ;

:test/
	-7 ( 7 <? )(
		dup 5 * "%d 8 / =" print
		dup 5 * 8 / "%d " print cr
		1+ ) drop ;

:testmod
	-7 ( 7 <? )(
		dup "%d 5 mod =" print
		dup 5 mod "%d " print cr
		1+ ) drop ;


:testsh
	30 ( 37 <? )(
		dup "256 ffffffff %d *>> " print
		256 $fffffff pick2 *>> "=%h" print
		cr
		1+ ) drop ;


:main
	show clrscr
|        testdiv cr
|        test/mod cr
|        test/ cr
        testmod cr
|        testsh cr
		'exit >esc<
		;

: main ;
