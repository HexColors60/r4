:testdiv
	-7 ( 7 <? )(
		dup "%d 5 / =" print
		dup 5 / "%d " print cr
		1+ ) drop ;

:testmod
	-7 ( 7 <? )(
		dup "%d 5 mod =" print
		dup 5 mod "%d " print cr
		1+ ) drop ;

:testsh
	30 ( 37 <? )(
		dup "256 ffffffff %d *>> " print
		256 $fffffff pick2 *>> "=%h" print
		cr
		1+ ) drop ;


:main
	show clrscr
        testdiv cr
        testmod cr
|        testsh cr
		'exit >esc<
		;

: main ;
