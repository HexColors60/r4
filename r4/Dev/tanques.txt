|tanque1
#xt -0.5 #yt #dt 0 #vt #adt
#xb #yb #db #vxb #vyb #tb
#danio

|tanque2
#xt2 0.5 #yt2 #dt2 0.5 #vt2 #adt2
#xb2 #yb2 #db2 #vxb2 #vyb2 #tb2
#danio2


#tanques 0 0
#piedras 0 0
#fx 0 0

|-----------------
:r0001 rand 0.001 mod ;
:r001 rand 0.01 mod ;
:r01 rand 0.1 mod ;

:fpos
	8 >> sh 2/ + swap 8 >> sw 2/ + swap pos ;

:distancia
	 rot - dup *. >r - dup *. r> + sqrt. ;

|-----------------
:phumo
	dup @ 1- 0? ( nip ; )
	dup 2* alpha
	swap !+ >r
	r@+	r@+ fpos
	r@+ r 12 - +!	| VX
	r@+ r 12 - +!	| VY
	r@+ qdim
	r@+ r@+ rnsprite
	r@+ r> 8 - +!
	255 alpha
	;

:+humo | x y --
	'phumo 'fx p!+ >r
	127 r!+ | vida
	r!+ r!+ | x y
	r0001 r!+ r0001 r!+ | vx vy
	r0001 4 + r!+		| tamaño
	'humo2 r!+			| dibujo
	0 r!+
	r001 r!+
	rdrop ;

:disparo
	tb 1? ( drop ; ) drop
	dup >r
	r@+ 'xb ! r@+ 'yb ! r> 4+ @ 'db !
	db sincos 0.01 *. 'vxb ! 0.01 *. 'vyb !
	175 'tb ! ;

#tand 0 0 tand1 10 tand2 10 tand3 10 0
#tani 0 0 tani1 10 tani2 10 tani3 10 0

:secani | 'secani -- 'spr
	>r r @ 0? ( r 8 + r!+ r!+ r> @ ; )
	r 4+ @ 1+ over 4+ @ | 'spr cnt max
	<? ( r> 4+ ! @ ; ) drop
	8 + dup @ 0? ( 2drop r 8 + dup @ ) | adr spr
	swap r!+ 0 r> ! ;

:oruga
	over
	0? ( drop
		'tand1 over rnsprite
		'tani1 over rnsprite ; )
	drop
	'tand secani over rnsprite
	'tani secani over rnsprite
	;

:drawtan | rot --
	oruga
	'tanc over rnsprite
	'tant over rnsprite
 	;

:drawtan2 | rot --
	oruga
	'tanc2 over rnsprite
	'tant over rnsprite
 	;

| tanque 1
| x y dibujo dt
| (adt) (vt)
:t1
	>r
	80 qdim
	r@+	r@+ fpos
	r@+ drop
	|r@+ drawtan  |rnsprite
	r> 4+
	[ -0.001 over ! ; ] <le>
	[ 0.001 over ! ; ] <ri>
	[ 0 over ! ; ] >le<
	[ 0 over ! ; ] >ri<
	4+
	[ 0.001 over ! ; ] <up>
	[ 0 over ! ; ] >up<
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ drawtan | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
	r> 20 -
	'disparo <dn>
	drop
	;

:disparo2
	tb2 1? ( drop ; ) drop
	dup >r
	r@+ 'xb2 ! r@+ 'yb2 ! r> 4+ @ 'db2 !
	db2 sincos 0.01 *. 'vxb2 ! 0.01 *. 'vyb2 !
	175 'tb2 ! ;

:t2
	>r
	80 qdim
	r@+	r@+ fpos
	r@+ drop
	|r@+ drawtan2 |rnsprite
	r> 4+
	[ -0.001 over ! ; ] 30 ?key
	[ 0.001 over ! ; ] 32 ?key
	[ 0 over ! ; ] 30 ?ukey
	[ 0 over ! ; ] 32 ?ukey
	4+
	[ 0.001 over ! ; ] 17 ?key
	[ 0 over ! ; ] 17 ?ukey
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ drawtan2 | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
	r> 20 -
	'disparo2 31 ?key
	drop
	;

| tanque de prueba robot
:t3
	>r
	80 qdim
	r@+	r@+ fpos
	r@+ r@+ rnsprite
	r>
	[ -0.001 over ! ; ] 30 ?key
	[ 0.001 over ! ; ] 32 ?key
	[ 0 over ! ; ] 30 ?ukey
	[ 0 over ! ; ] 32 ?ukey
	4+
	[ 0.001 over ! ; ] 17 ?key
	[ 0 over ! ; ] 17 ?ukey
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
	r> 20 -
	'disparo2 31 ?key
	drop
	;


:reset
	'tanques p.clear
	'piedras p.clear
	'fx p.clear

	't1 'tanques p!+ >r
	0 0 0 0 'stanque 0 -0.5 r!+ r!+ r!+ r!+ r!+ r!+
	0 r!+
	rdrop

	't2 'tanques p!+ >r
	0 0 0 0.5 'stanque 0 0.5 r!+ r!+ r!+ r!+ r!+ r!+
	0 r!+
	rdrop

	;

|--------------
:ganotanque2
	show clrscr
		"gano tanque2!" print
		'exit >esc< ;

:ganotanque1
	show clrscr
		"gano tanque1!" print
		'exit >esc< ;

:golpet1
	0 'tb2 ! 0 'vt !
	10 'danio +!
	danio 100 >=? ( ganotanque2 exit ) drop
	10 ( 1? )( yt xt +humo 1- ) drop
	;

:golpet2
	0 'tb ! 0 'vt2 !
	10 'danio2 +!
	danio2 100 >=? ( ganotanque1 exit ) drop
	10 ( 1? )( yt2 xt2 +humo 1- ) drop
	;


:bala1
	tb 0? ( drop ; ) drop
	48 qdim xb yb fpos
	'misil db rnsprite

	vxb 'xb +! vyb 'yb +! -1 'tb +!

	xb yb xt2 yt2 distancia
	0.05 <? ( golpet2 ) drop
	;

:bala2
	tb2 0? ( drop ; ) drop
	48 qdim xb2 yb2 fpos
	'misil db2 rnsprite

	vxb2 'xb2 +! vyb2 'yb2 +! -1 'tb2 +!

	xb2 yb2 xt yt distancia
	0.05 <? ( golpet1 ) drop
	;


:main
	reset
	$3f3f paper
	show clrscr
		"tanques " print
		danio "t1: %d " print
		danio2 "t2: %d" print

		bala1
		bala2

		'tanques p.draw
		'fx p.draw

		'exit >esc<
		cminiflecha ;

:memoria
	mark
	10 'tanques p.ini
	100 'piedras p.ini
	1000 'fx p.ini
	;

: memoria main ;
