|----- data
:gendata | adr --
	dup 8 + @
	1 nand? ( 2drop ; )
	drop

	@ "#%w" slog
	;

::r3-gendata
	dicc ( dicc> <? )(
		dup gendata
		16 + ) drop ;

|----- code
:gencode | adr --
	dup 8 + @
	1 and? ( 2drop ; )
	12 >> $fff and
	0? ( 2drop ; ) drop | no hay llamadas
	@ ":%w" slog
	;

::r3-gencode
	dicc ( dicc> <? )(
		dup gencode
		16 + ) drop ;
