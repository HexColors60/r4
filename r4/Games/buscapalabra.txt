#tiempo
#cntpal
#lista
#elegida
#pantalla
#puntos

#palabras
"uno"
"dos"
"tres"
"cuatro"
"cinco"
"seis"
"siete"
"ocho"
"nueve"
"diez"
0

:generalista
	0 'cntpal !
	'palabras
	( dup c@ 1? )( drop
		dup , 1 'cntpal +!
		( c@+ 31 >? )( drop ) drop
		trim ) 2drop ;

:memoria
	mark
	here 'lista !
	generalista
	here 'pantalla ! ;

:nropal |4 -5
	4 * lista + @ "%l" mprint ;

:ppal | x y p --
	nropal
	sizeprint cch
	rot >r
	neg 2/ rot + swap
	neg 2/ rot +	swap
	over 1- over 1- 
	atxy r ink@ >r blanco print 
	atxy r> ink r> print
	;
	
	

:inipant
	pantalla >r
	0 ( cntpal <? )( 
		sw 2/ rnd $ff and 128 - + r!+
		sh 2/ rnd $ff and 128 - + r!+
		dup r!+
		rand r!+ 
		1+ ) drop
	0 r> ! 
	rnd cntpal mod abs 'elegida ! 	
	;

:continua
  inipant
  1 'puntos +!
  ;

:inppal
	nropal
	sizeprint cch
	rot >r
	neg 2/ rot +
	swap
	neg 2/ rot +
	swap atxy
	r> 
	print2gc drop
	'continua guiBtn
	;

:clickelegida
	elegida 16 * pantalla + >r
	r@+ r@+ r@+ rdrop
	inppal ;

:verpant
	pantalla >r 
	( r@+ 1? )( r@+ r@+ r@+ ink ppal )
	drop rdrop ; 

:movpant
	pantalla >r
	( r@+ 1? )( drop
		rnd sign r 4 - +! drop 
		rnd sign r +! drop
		12 r+ )
	drop rdrop
	;
	
:main
	0 'puntos !
	60 'tiempo ! 
	inipant
	show clrscr
		robotobold 0.11 %s rfont! 
		home blanco
		0 0 gotoxy elegida nropal printc
|		inpal "%d" prin
		0 rows 1- gotoxy puntos tiempo "%d  %d" printc 

		robotobold 0.09 %s rfont! 
		verpant
		movpant
		clickelegida		
		
		'inipant <f1>
		'exit >esc<
		cminiflecha 
		1 .segs .restart
		-1 'tiempo +! 
		tiempo 0? ( exit ) drop ;

: memoria main ;
