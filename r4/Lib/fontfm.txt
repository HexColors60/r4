^r4/lib/print.txt


#col 0 $555555 $aaaaaa $ffffff


:charsizem | byte -- size
	ccw nip ;

:charline | sx n bit --
	ccw ( 1? )( 1- swap dup $3 and 2 << 'col + @ px!+ 2 >> swap ) 2drop ;

:charm | c --
	ccx ccy setxy
	charlin * charrom +  >b
	sw ccw -
	cch ( 1? )( 1-
		b@+ charline
		over px+!
		) 2drop ;

::fontm | 'fontm --
	>a a@+ dup 2 << 'charlin !
	a@+ swap fixsize
	a> 'charrom !
	'charm 'charsizem setfont ;

::setcolor | front back --
	'col >r
	dup r!+
	2dup color33% r!+
	over color33% r!+
	r> ! ;
