:shader
	@+ $7fffffff =? ( drop 1 px+! 4+ ; )
	8 >> $f and 1+
	swap @+ $f0f0f0 and
	rot *
	px!+ ;

:shader
	@+ $7fffffff =? ( drop 1 px+! 4+ ; )
	swap @+ 4 >> $f0f0f and
	rot 
	pick2 @ - 7 + 15 clamp0max 1+
	* $f0f0f0 and
	dup 4 >> or
	px!+ ;

::zdraw | x y --
	setxy
	sw w3do -
	zb
	h3do ( 1? )( 1-
		w3do ( 1? )( 1-
			rot shader rot rot
			) drop
		pick2 px+!
		) 3drop ;

:freelook
	xymouse
	sh 2/ - 7 << swap
	sw 2/ - neg 7 << swap
	neg mrotx mroty
	;

|-------------
#sizx #sizy #sizz

:drawz | z --
	sizx neg sizy neg pick2 3dop
	sizx sizy neg pick2 3dline
	sizx sizy pick2 3dline
	sizx neg sizy pick2 3dline
	sizx neg sizy neg rot 3dline ;

:drawl | x1 x2 --
	2dup sizz neg 3dop sizz 3dline ;

:box3d | sx sy sz --
	'sizz ! 'sizy ! 'sizx !
	sizz neg drawz sizz drawz
	sizx neg sizy neg drawl sizx sizy neg drawl
	sizx sizy drawl sizx neg sizy drawl
	;

:dpoint | nro val -- nro
	8 >> sh 1- clamp0max
	over swap setxy $ff0000 px!+
	;

	zb nline w3do * 3 << +
	>r
	0 ( sw <? )(
     	r@+ dpoint
     	4 r+
		1+ ) drop
	rdrop
	;
