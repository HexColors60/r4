|--
| x cx y cy z dxy col ccol
| 0 4 8 12 16 20  24  28

:samecuadrant | adr -- adr 1/0-si
	dup >a
	a@+ a@+ - $80000000 and a@+ a@+ - 2/ $40000000 and or 16 a+
	a@+ a@+ - $80000000 and a@+ a@+ - 2/ $40000000 and or 16 a+
	<>? ( drop 1 ; )
	a@+ a@+ - $80000000 and a@+ a@+ - 2/ $40000000 and or
	- ;


| x cx y cy z dxy col ccol
| 0 4 8 12 16 20  24  28
| cx>x --> cx-=z : cx+=z dx+=256 ; z=z/2
:advvert
	a> 16 + @	| z
	a@+ a@ swap | cx x
	>? ( over - a!+ )( over + a!+ pick2 a> 12 + +! )
	a@+ a@ swap
	>? ( over - a!+ )( over + a!+ pick2 9 << a> 4+ +! )
	2/ a!+		| z=z/2
	12 a+ ;

:advancecuadrant | adr -- adr
	dup >a
	advvert advvert advvert ;

:cpyvert | --	a:o b:d
	a@+ b!+ a@+ b!+ a@+ b!+ a@+ b!+
	a@+ b!+ a@+ b!+ a@+ b!+ a@+ b!+
	;

:2/1 | o1 o2 -- om ;
|	+ 2/ ;
	over - dup 31 >> - 2/ + ;

:avgvert | o1 --	a:o2 b:d
	@+ a@+ 2/1 b!+ @+ a@+ 2/1 b!+
	@+ a@+ 2/1 b!+ @+ a@+ 2/1 b!+
	@+ a@+ 2/1 b!+ @+ a@+ nip |+ 2/
	b!+
	@+ a@+ 2/1 b!+ @+ a@+ 2/1 b!+
	drop
	;

:avglev | o1 --
	dup avgvert dup avgvert avgvert ;

:verxy
	@+ swap 4+ @+ swap 4+ @
	3dpro ;

:verxy
	@+ swap 4+ @+ swap 4+ @
	project3d  ;

:printv
	@+ "x:%d " print
	@+ "xc:%d " print
	@+ "y:%d " print
	@+ "yc:%d " print
	@+ "z:%d " print
	@+ "dxy:%h " print
	@+ "col:%h " print
	@+ drop |"dcol:%h " print
	cr
	allowchome
	;

:testcua | adr
	dup >b
	b@+ b@+ - b@+ b@+ - | (x<cx -) (y<cy -)
	"%d %d " print
	16 b+
	b@+ b@+ - b@+ b@+ -
	"%d %d " print
	16 b+
	b@+ b@+ - b@+ b@+ -
	"%d %d " print
	;


:dump | stakc level --
	clrscr
	verde
	dup "%d " print cr
	stackv ( pick2 1+ <? )(
		blanco samecuadrant "%d - " print
		testcua
		rand ink
    	dup verxy >r >r
		printv
		dup verxy >r >r
		printv
		dup verxy 2dup op r> r> line r> r> line line
		printv
		) drop
|	stackv ( pick2 <? )(
|		dup verxy >r >r 32 + dup verxy >r >r 32 + dup verxy 2dup op r> r> line r> r> line line
|		32 + ) drop
	show 'exit >esc<
	;
